#include <iostream>
#include <algorithm>
#include <cstring>
using namespace std;

struct Node {
    int childID;
    int v, w;
};
struct CNode {
    int next;
    int id;
};

void dfs(Node n[], CNode e[], int *dp, int V, int root) {
    int son = n[root].childID;
    for (int j = V - 1; j >= n[root].v; --j) {
        dp[root * V + j] = n[root].w;
    }
    while (son != -1) {
        dfs(n, e, dp, V, e[son].id);
        for (int j = V - 1; j >= n[root].v; --j) {
            int limit = j - n[root].v;
            for (int k = 0; k <= limit; ++k) {
                dp[root * V + j] = max(dp[root * V + j], dp[e[son].id * V + k] + dp[root * V + j - k]);
            }
        }
        son = e[son].next;
    }
}

int main() {
    int N, V;
    cin >> N >> V;
    Node n[N];
    CNode e[N - 1];
    int idx = 0;
    memset(n, -1, sizeof(n));
    memset(e, -1, sizeof(e));
    int root, tv, tw, tp;
    for (int i = 0; i < N; ++i) {
        cin >> tv >> tw >> tp;
        n[i].v = tv, n[i].w = tw;
        if (tp == -1) root = i;
        else e[idx].next = n[tp - 1].childID, e[idx].id = i, n[tp - 1].childID = idx++;
    }
    int dp[N][V + 1];
    memset(dp, 0, sizeof(dp));
    dfs(n, e, &dp[0][0], V + 1, root);
    cout << dp[root][V] << endl;
    return 0;
}
